def getWeather():
    city=textfield.get()


    geolocator=Nominatim(user_agent="geoapiExercises")
    location=geolocator.geocode(city)
    obj=TimezoneFinder()
    result=obj.timezone_at(lng=location.longitude, lat=location.latitude)
    timezone.config(text=result)
    long_lat.config(text=f"{round(location.latitude,4)}°N,{round(location.longitude,4)}°E")
    home=pytz.timezone(result)
    local_time=datetime.now(home)
    current_time=local_time.strftime("%I:%M%p")
    clock.config(text=current_time)

    #weather
    api = "https://api.openweathermap.org/data/2.5/weather?lat="+str(location.latitude)+"&lon="+str(location.longitude)+"&appid=535f4937ec1f772806bdfd28a8155be1&units=metric"
    json_data=requests.get(api).json()

    #current

    temp=json_data['main']['temp']
    humidity=json_data["main"]["humidity"]
    pressure = json_data["main"]["pressure"]
    wind=json_data["wind"]["speed"]
    description=json_data["weather"][0]["description"]

    t.config(text=(temp,"°C"))
    h.config(text=(humidity,"%"))
    p.config(text=(pressure, "hPa"))
    w.config(text=(wind, "m/s"))
    d.config(text=description)


    #first cell
    api2 = "https://api.openweathermap.org/data/2.5/forecast?q="+city+"&appid=535f4937ec1f772806bdfd28a8155be1"
    json_data2 = requests.get(api2).json()

    date = json_data2['list'][0]["dt_txt"]
    timenow = date[11:13]
    timenowi = int(timenow)
    tmp = 0
    tmpd=0
    tmp5 = 0
    tmpd5=0

    if timenowi==0: #if time now is 00:00
        tmpd=0 #day temperature
        tmp = 7 #night temperature
        tmpd5=0 #temperature today
        tmp5 = 7 #temperature tonight

    if timenowi==3: #if time now is 03:00
        tmpd = -1
        tmp=6
        tmpd5=0
        tmp5 = 6

    if timenowi==6:  #if time now is 06:00
        tmpd = -2
        tmp=5
        tmpd5=0
        tmp5 = 5

    if timenowi==9: #if time now is 09:00
        tmpd = -3
        tmp=4
        tmpd5=0
        tmp5 = 4

    if timenowi==12: #if time now is 12:00
        tmpd = -4
        tmp=3
        tmpd5=0
        tmp5 = 3

    if timenowi==15: #if time now is 15:00
        tmpd = -5
        tmp=2
        tmpd5=0
        tmp5 = 2

    if timenowi==18: #if time now is 18:00
        tmpd = -6
        tmp=1
        tmpd5=0
        tmp5 = 1

    if timenowi==21: #if time now is 21:00
        tmpd = -7
        tmp=0
        tmpd5=0
        tmp5 = 0

    firstdayimage=json_data["weather"][0]["icon"]
    date2 = json_data2['list'][0]["dt_txt"]
    #photo1=ImageTk.PhotoImage(file=f"C:/Users/Valya/Desktop/Do2/Python/weather forecast/icon3/{firstdayimage}@2x.png")
    photo1=ImageTk.PhotoImage(file=f"icon3/{firstdayimage}@2x.png")
    firstimage.config(image=photo1)
    firstimage.image=photo1

    temp1=json_data2['list'][tmp+8]['main']['temp']
    temp1d=json_data2['list'][tmpd+8]['main']['temp']
    Degrees1 = str(round(temp1 - 273.15))
    Degreesd1 = str(round(temp1d - 273.15))
    day2temp.config(text=f"Day:{Degreesd1}\n Night: {Degrees1}")

    #second cell
    seconddayimage=json_data2['list'][8]["weather"][0]["icon"]

    #img=(Image.open(f"C:/Users/Valya/Desktop/Do2/Python/weather forecast/icon3/{seconddayimage}@2x.png"))
    img = (Image.open(f"icon3/{seconddayimage}@2x.png"))
    resized_image=img.resize((50,50))
    photo2=ImageTk.PhotoImage(resized_image)
    secondimage.config(image=photo2)
    secondimage.image=photo2


    temp2=json_data2['list'][tmp+16]['main']['temp']
    temp2d=json_data2['list'][tmpd+16]['main']['temp']
    Degrees2 = str(round(temp2 - 273.15))
    Degreesd2 = str(round(temp2d - 273.15))
    day3temp.config(text=f"Day:{Degreesd2}\n Night: {Degrees2}")

    #
    # #third cell
    thirddayimage=json_data2['list'][16]["weather"][0]["icon"]

    #img=(Image.open(f"C:/Users/Valya/Desktop/Do2/Python/weather forecast/icon3/{thirddayimage}@2x.png"))
    img = (Image.open(f"icon3/{thirddayimage}@2x.png"))
    resized_image=img.resize((50,50))
    photo3=ImageTk.PhotoImage(resized_image)
    thirdimage.config(image=photo3)
    thirdimage.image=photo3
    #

    temp3 = json_data2['list'][tmp + 24]['main']['temp']
    temp3d = json_data2['list'][tmpd + 24]['main']['temp']
    Degrees3 = str(round(temp3 - 273.15))
    Degreesd3 = str(round(temp3d - 273.15))
    day4temp.config(text=f"Day:{Degreesd3}\n Night: {Degrees3}")

    # #fourth cell
    fourthdayimage=json_data2['list'][24]["weather"][0]["icon"]

    #img=(Image.open(f"C:/Users/Valya/Desktop/Do2/Python/weather forecast/icon3/{fourthdayimage}@2x.png"))
    img = (Image.open(f"icon3/{fourthdayimage}@2x.png"))
    resized_image=img.resize((50,50))
    photo4=ImageTk.PhotoImage(resized_image)
    fouthimage.config(image=photo4)
    fouthimage.image=photo4

    temp4 = json_data2['list'][tmp + 32]['main']['temp']
    temp4d = json_data2['list'][tmpd + 32]['main']['temp']
    Degrees4 = str(round(temp4 - 273.15))
    Degreesd4 = str(round(temp4d - 273.15))
    day5temp.config(text=f"Day:{Degreesd4}\n Night: {Degrees4}")

    #
    #fifth cell
    fifthdayimage=json_data2['list'][0]["weather"][0]["icon"]

    #img=(Image.open(f"C:/Users/Valya/Desktop/Do2/Python/weather forecast/icon3/{fifthdayimage}@2x.png"))
    img = (Image.open(f"icon3/{fifthdayimage}@2x.png"))
    resized_image=img.resize((50,50))
    photo5=ImageTk.PhotoImage(resized_image)
    fifthimage.config(image=photo5)
    fifthimage.image=photo5

    temp0=json_data2['list'][tmp5]['main']['temp']
    temp0d=json_data2['list'][tmpd5]['main']['temp']
    Degrees0 = str(round(temp0 - 273.15))
    Degreesd0 = str(round(temp0d - 273.15))
    day1temp.config(text=f"Day:{Degreesd0}\n Night: {Degreesd0}")

    #days

    first = datetime.now()
    day1.config(text=first.strftime("%A"))

    second=first+timedelta(days=1)
    day2.config(text=second.strftime("%A"))

    third=first+timedelta(days=2)
    day3.config(text=third.strftime("%A"))

    fourth=first+timedelta(days=3)
    day4.config(text=fourth.strftime("%A"))

    fifth=first+timedelta(days=4)
    day5.config(text=fifth.strftime("%A"))
